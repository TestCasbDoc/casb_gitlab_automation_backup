import requests, zipfile, tempfile, os

def upload_to_server(run_folder, server="http://SERVER_IP:4012"):
    tmp = tempfile.mktemp(suffix=".zip")
    with zipfile.ZipFile(tmp, "w", zipfile.ZIP_DEFLATED) as zf:
        run_name = os.path.basename(run_folder)
        for root, dirs, files in os.walk(run_folder):
            for file in files:
                abs_path = os.path.join(root, file)
                arc_name = os.path.join(run_name,
                    os.path.relpath(abs_path, run_folder))
                zf.write(abs_path, arc_name)
    with open(tmp, "rb") as f:
        r = requests.post(f"{server}/api/upload",
                          files={"file": (run_name+".zip", f)})
    print(f"[UPLOAD] {r.json()}")
    os.remove(tmp)
